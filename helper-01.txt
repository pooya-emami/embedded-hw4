# Test all sensors in one go
for sensor in "temperature:101" "co2:204" "smoke:304"; do
    type=$(echo $sensor | cut -d: -f1)
    id=$(echo $sensor | cut -d: -f2)
    echo "=== Testing $type $id ==="
    curl -s "http://192.168.233.139:8080/query?sensor_type=$type&sensor_id=$id"
    echo -e "\n"
done
